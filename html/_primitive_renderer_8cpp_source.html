165bool PrimitiveRenderer::isLinesCrossing(LineSegment line1, LineSegment line2) {
166 Point2D line1StartPoint = line1.getStartPoint();
167 Point2D line2StartPoint = line2.getStartPoint();
168 Point2D line1EndPoint = line1.getEndPoint();
169 Point2D line2EndPoint = line2.getEndPoint();
170 float dx1 = line1EndPoint.getX() - line1StartPoint.getX();
171 float dx2 = line2EndPoint.getX() - line2StartPoint.getX();
172 float dy1 = line1EndPoint.getY() - line1StartPoint.getY();
173 float dy2 = line2EndPoint.getY() - line2StartPoint.getY();
174 float cross = dx1 * dy2 - dx2 * dy1;
175 if (cross == 0)
176 return false;
177 else {
178 float crossX = (((line2StartPoint.getX() - line1StartPoint.getX()) * dy2 -
179 (line2StartPoint.getY() - line1StartPoint.getY()) * dx2) / cross);
180 float crossY = (((line2StartPoint.getX() - line1StartPoint.getX()) * dy1 -
181 (line2StartPoint.getY() - line1StartPoint.getY()) * dx1) / cross);
182 if (crossX > 0 && crossX < 1 && crossY > 0 && crossY < 1)
183 return true;
184 }
185 return false;
186}
